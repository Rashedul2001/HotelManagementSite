
// Room Type dropdown 
using Microsoft.AspNetCore.Mvc;
using HotelManagementSite.Models.Domain;

public class HotelRoomController : Controller
{
    public IActionResult Create()
    {
        ViewBag.RoomTypes = RoomTypes.Types; // Pass the list of room types to the view
        return View();
    }

    [HttpPost]
    public IActionResult Create(HotelRoom hotelRoom)
    {
        if (ModelState.IsValid)
        {
            // Save the hotelRoom to the database
            return RedirectToAction("Index");
        }

        ViewBag.RoomTypes = RoomTypes.Types; // Re-populate the list if validation fails
        return View(hotelRoom);
    }
}
@model HotelManagementSite.Models.Domain.HotelRoom

<form asp-action="Create" method="post">
    <div>
        <label for="Name">Room Name</label>
        <input type="text" id="Name" name="Name" class="form-control" value="@Model.Name" required />
    </div>

    <div>
        <label for="RoomType">Room Type</label>
        <select id="RoomType" name="RoomType" class="form-control" required>
            <option value="">-- Select Room Type --</option>
            @foreach (var type in ViewBag.RoomTypes as List<string>)
            {
                <option value="@type" @(type == Model.RoomType ? "selected" : "")>@type</option>
            }
        </select>
    </div>

    <button type="submit" class="btn btn-primary">Submit</button>
</form>
